### 性能测试脚本
```bash
#!/bin/bash
# test-performance.sh

echo "🧪 开始性能测试..."

# 测试响应时间
echo "测试响应时间..."
for i in {1..100}; do
  START=$(date +%s%N)
  # 执行简单查询
  END=$(date +%s%N)
  ELAPSED=$(( (END - START) / 1000000 ))
  echo "$ELAPSED" >> /tmp/response_times.log
done

AVG=$(awk '{sum+=$1} END {print sum/NR}' /tmp/response_times.log)
echo "平均响应时间：${AVG}ms"

# 测试缓存命中率
echo "测试缓存命中率..."
# 分析缓存日志
```
